-- 4)Crie uma tabela para armazenar informações de produtos Devem ter os seguintes atributos:
--   •Código de identificação; Nome curto, Nome descritivo, Data da criação, Local no estoque, Quantidade e Preço.
CREATE TABLE Produto(
	IDProduto int IDENTITY,
	NomeCurto VARCHAR(20) NOT NULL,
	NomeDescritivo VARCHAR(40) NOT NULL,
	DataDeCriação DATETIME DEFAULT GETDATE(),
	LocalNoEstoque VARCHAR(40) NOT NULL,
	Quantidade int NOT NULL,
	Preço DECIMAL(8,2) NOT NULL,
		CONSTRAINT PK_IDProduto PRIMARY KEY(IDProduto),
		CONSTRAINT UK_NomeCurto UNIQUE(NomeCurto)
);
